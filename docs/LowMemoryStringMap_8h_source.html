367 static void addTo(Branch* branch, const StringT& str, size_t idx, const ValueT& val)
368 {
369 CharT c(0);
370 if (idx < str.size())
371 {
372 c = str[idx];
373 }
374
375 auto it = branch->m_children.find(c);
376 if (it == branch->m_children.end())
377 {
378 size_t newIdx = idx + 1 >= str.size() ? str.size() : idx + 1;
379 branch->m_children.emplace(c, branch->createLeaf(str.substr(newIdx), val));
380 return;
381 }
382
383 idx++;
384
385 std::unique_ptr<Node> child = std::move(it->second);
386 StringT childStr = child->getString();
387
388 auto p = std::mismatch(childStr.begin(), childStr.end(), str.begin() + idx, str.end());
389
390 if (p.first == childStr.end() && p.second == str.end())
391 {
392 // adding same string, abort
393 it->second = std::move(child);
394 return;
395 }
396
397 size_t length = std::distance(childStr.begin(), p.first);
398
399 std::unique_ptr<Branch> newBranch;
400 if (!dynamic_cast<Branch*>(child.get()) || p.first != childStr.end())
401 {
402 newBranch = branch->split(std::move(child), length);
403 }
404 else
405 {
406 newBranch = std::unique_ptr<Branch>(dynamic_cast<Branch*>(child.release()));
407 }
408
409 addTo(newBranch.get(), str, idx + length, val);
410
411 it->second = std::move(newBranch);
412 }

415 {
416 size_t s = m_children.size() * sizeof(std::pair<CharT, std::unique_ptr<Node>>);

476 std::unique_ptr<Branch> split(std::unique_ptr<Node> node, size_t idx) const
477 {
478 StringT str = node->getString();
479 std::unique_ptr<Branch> frontBranch = createBranch(str.substr(0, idx));
480
481 CharT c(0);
482 if (idx < str.size())
483 {
484 c = str[idx];
485 idx++;
486 }
487
488 Branch* oldBranch = dynamic_cast<Branch*>(node.get());
489 if (oldBranch)
490 {
491 std::unique_ptr<Branch> backBranch = createBranch(str.substr(idx));
492 backBranch->m_children = std::move(oldBranch->m_children);
493 frontBranch->m_children.emplace(c, std::move(backBranch));
