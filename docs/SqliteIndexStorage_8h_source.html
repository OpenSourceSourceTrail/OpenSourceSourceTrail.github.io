277 public:
278 void compile(
279 const std::string header,
280 size_t valueCount,
281 std::function<void(CppSQLite3Statement& stmt, const StorageType&, size_t)> bindValuesFunc,
282 CppSQLite3DB& database)
283 {
284 m_bindValuesFunc = bindValuesFunc;
285
286 std::string valueStr = '(' +
287 utility::join(std::vector<std::string>(valueCount, "?"), ',') + ')';
288
289 const size_t MAX_VARIABLE_COUNT = 999;
290 size_t batchSize = MAX_VARIABLE_COUNT / valueCount;
291
292 while (true)
293 {
294 std::stringstream stmt;
295 stmt << header;
296
297 for (size_t i = 0; i < batchSize; i++)
298 {
299 if (i != 0)
300 {
301 stmt << ',';
302 }
303 stmt << valueStr;
304 }
305 stmt << ';';
306
307 m_stmts.emplace_back(
308 std::make_pair(batchSize, database.compileStatement(stmt.str().c_str())));
309
310 if (batchSize == 1)
311 {
312 break;
313 }
314 else
315 {
316 batchSize /= 2;
317 }
318 }
319 }
320
321 bool execute(const std::vector<StorageType>& types, SqliteIndexStorage* storage)
322 {
323 size_t i = 0;
324 for (std::pair<size_t, CppSQLite3Statement>& p: m_stmts)
325 {
326 const size_t& batchSize = p.first;
327 CppSQLite3Statement& stmt = p.second;
328
329 while (types.size() - i >= batchSize)
330 {
331 for (size_t j = 0; j < batchSize; j++)
332 {
333 m_bindValuesFunc(stmt, types[i + j], j);
334 }
335
336 const bool success = storage->executeStatement(stmt);
337 if (!success)
338 {
339 return false;
340 }
341
342 i += batchSize;
343 }
344 }
345
346 return true;
347 }

349 private:
350 std::vector<std::pair<size_t, CppSQLite3Statement>> m_stmts;
