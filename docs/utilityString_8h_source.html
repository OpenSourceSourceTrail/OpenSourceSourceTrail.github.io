105ContainerType split(const std::string& str, const std::string& delimiter)
106{
107 size_t pos = 0;
108 size_t oldPos = 0;
109 ContainerType c;
110
111 do
112 {
113 pos = str.find(delimiter, oldPos);
114 c.push_back(str.substr(oldPos, pos - oldPos));
115 oldPos = pos + delimiter.size();
